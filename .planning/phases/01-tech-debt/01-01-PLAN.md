```java
// 在 handle() 或 fill() 方法开始处
String actualKey = context.getActualKey();
Object existingValue = cache.get(actualKey); // 获取当前值

if (existingValue instanceof CachedValue) {
    CachedValue cached = (CachedValue) existingValue;
    long remainingTtl = cached.getRemainingTtl();
    if (remainingTtl >= 0 && remainingTtl < refreshAheadThresholdMs) {
        // TTL 即将过期，跳过填充
        log.debug("Skipping pre-refresh for key {}: remaining TTL={}ms", actualKey, remainingTtl);
        return;
    }
}
```
